# Kokkai バッチ検索用サンプルファイル
# 1行に1クエリ分の CLI 引数を記述
# # で始まる行はコメント

# 生成AI関連（最近）
--query "生成AI" --speaker "岸田" --from "2023-01-01"

# デジタル関連（古い期間まで）
--query "デジタル" -s "上野" --from "1989年11月01日まで"

# AI規制（複数年）
--query "AI規制" --from "2022年" --until "2025年12月31日"

# 追加サンプル
--query "生成AI" --limit 5
--query "データ" --speaker "山田" --from "2024-01-01" --until "2024-12-31"
